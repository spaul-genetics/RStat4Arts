install.packages('ggplot2')
library(ggplot2)

ggplot(data = iris)+
  aes(x = Sepal.Length, y = Sepal.Width)+
  geom_point()

Color points

ggplot(data = iris)+
  aes(x = Sepal.Length, y = Sepal.Width, color = Species)+

Theme

ggplot(data = iris)+
  aes(x = Sepal.Length, y = Sepal.Width, color = Species)+
  geom_point()+
  theme_bw()+
  labs(x = 'Sepal Length', y = 'Sepal Width')+
  theme(legend.position = 'top', 
        axis.title = element_text(face = 'bold'))
